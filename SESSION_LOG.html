export function slugify(text: string): string {
  return text
    .toLowerCase()
    .replace(/[^a-z0-9\s-]/g, '')
    .trim()
    .replace(/\s+/g, '-')
    .replace(/-+/g, '-')
    .slice(0, 80)
}

Write: lib/supabase.ts (final version — lazy Proxy)

import { createClient, SupabaseClient } from '@supabase/supabase-js'

let _public: SupabaseClient | null = null
let _admin: SupabaseClient | null = null

function getPublic(): SupabaseClient {
  if (!_public) {
    const url = process.env.NEXT_PUBLIC_SUPABASE_URL
    const key = process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY
    if (!url || !key) throw new Error('Missing NEXT_PUBLIC_SUPABASE_URL or NEXT_PUBLIC_SUPABASE_ANON_KEY')
    _public = createClient(url, key)
  }
  return _public
}

function getAdmin(): SupabaseClient {
  if (!_admin) {
    const url = process.env.NEXT_PUBLIC_SUPABASE_URL
    const key = process.env.SUPABASE_SERVICE_ROLE_KEY
    if (!url || !key) throw new Error('Missing NEXT_PUBLIC_SUPABASE_URL or SUPABASE_SERVICE_ROLE_KEY')
    _admin = createClient(url, key)
  }
  return _admin
}

export const supabase = new Proxy({} as SupabaseClient, {
  get(_target, prop) {
    const client = getPublic()
    const value = (client as unknown as Record<string | symbol, unknown>)[prop]
    return typeof value === 'function' ? value.bind(client) : value
  },
})

export const supabaseAdmin = new Proxy({} as SupabaseClient, {
  get(_target, prop) {
    const client = getAdmin()
    const value = (client as unknown as Record<string | symbol, unknown>)[prop]
    return typeof value === 'function' ? value.bind(client) : value
  },
})

What this project is

Scout Agent is an automated AI news digest pipeline for the AITrends.ng blog. Every 6 hours it: fetches the latest items from 10 RSS feeds (TechCrunch, VentureBeat, OpenAI, Google AI, HuggingFace, Hacker News, Simon Willison, DeepLearning.AI, Anthropic), deduplicates against a Supabase scout_memory table, groups fresh articles by category, sends each group to Gemini 2.5 Flash to generate a human-voice digest, builds a free cover image via Pollinations.ai, POSTs the digest to the AITrends.ng blog API, and fires a Slack notification to #aitrends-feed. Runs on GitHub Actions cron.

Problems found — in order of severity

1. Silent failures — the biggest issue

The GitHub Actions workflow had:

run: |
  timeout 110 node index.js || true

timeout 110 kills the process after 110 seconds. Fetching 10 RSS feeds + multiple Gemini calls + publishing + Slack can easily take 2–4 minutes. Scout may be dying mid-run every single time. || true means the workflow always reports green even when Scout crashed. No way to know if it's working.

2. index.js is wrong for GitHub Actions

index.js starts a node-cron daemon — fires runScout() once on startup, then schedules future runs every 6 hours. GitHub Actions already handles the scheduling and kills the process after timeout 110. The cron.schedule(...) line inside index.js will never fire. The workflow should run scout.js directly.

3. GitHub Actions secrets may not be set

The workflow references 7 secrets. If any haven't been added in GitHub → Settings → Secrets, every run fails with empty env vars — and || true hides it.

4. Wrong Supabase key name in workflow

memory.js uses process.env.SUPABASE_SECRET_KEY. The workflow also passed SUPABASE_PUBLISHABLE_KEY which nothing in the code uses — dead noise.

5. hasSeen() fails open on Supabase errors

In memory.js:24-29, if Supabase is down, hasSeen returns false — treating every article as new. A Supabase outage would cause Scout to re-publish every article it has ever seen as duplicate posts.

6. No validation before publishing

If Gemini returns a malformed response, the parser could produce empty content or title. Scout would call publishPost with an empty body and the blog API would reject it with 400.

7. No failure alerts

Slack notification only fires on success. If the entire run fails, you get nothing.

8. API key name mismatch (works, but fragile)
